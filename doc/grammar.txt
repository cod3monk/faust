metacl  :
        | header in out
        | in out

header  : macrocall
        | header macrocall

in      : IN acl
        | IN

out     : OUT acl
        | OUT

acl     : rule
        | acl rule

rule    : action filter extensions
        | action filter

action  : PERMIT
        | DENY

filter      : PORTLESSPROTO HOST HOST               // example: IP, ICMP, ESP, AH
            | PORTFULLPROTO HOST port HOST port     // example: TCP, UDP
deprecated:            | PORTLESSPROTO6 HOST6 HOST6            // example: IP6, ICMP6
deprecated:            | PORTFULLPROTO6 HOST6 port HOST6 port  // example: TCP6, UDP6

port    :
        | PORT

extensions  : EXTENSION
            | extensions EXTENSION

macrocall   : MACRONAME ARGUMENTS
